## Anatomy

`Pagination` keeps the public surface small and exposes only styled building blocks.

```text
Pagination

   â”śâ”€ PaginationItem
   â”‚  â””â”€ PaginationLink

   â””â”€ PaginationItem
      â””â”€ PaginationNext
```

| Part                 | Role                                                               |
| -------------------- | ------------------------------------------------------------------ |
| `Pagination`         | Navigation landmark.                                               |
| `PaginationContent`  | Base UI toolbar root for keyboard navigation and item arrangement. |
| `PaginationItem`     | Lightweight wrapper for one pagination element.                    |
| `PaginationLink`     | Page link with active styling via `isActive`.                      |
| `PaginationPrevious` | Previous-page shortcut with a default icon-only control.           |
| `PaginationNext`     | Next-page shortcut with a default icon-only control.               |
| `PaginationEllipsis` | Non-interactive collapsed-range marker.                            |

## Composition

Pagination no longer owns page state, range generation, or URL mapping. Those concerns stay in
consumer code, where they are easier to reason about and easier to customize.

Use `usePagination` when you want the library to calculate the visible page items, ellipses, and
previous/next targets.

Use plain `href` when a normal anchor is enough. Use `render={<Link />}` when your app standardizes
on router link components. Use `render={<button />}` when pagination is driven by local state.

`PaginationPrevious` and `PaginationNext` are icon-only by default. Pass `children` when the
surrounding UI needs text labels or a custom layout.

The default examples keep a fixed pagination window, so the control does not grow when the current
page moves from the edge into the middle range.

## Links and Buttons

| Pattern                | Best fit                                              |
| ---------------------- | ----------------------------------------------------- |
| `href="/items?page=4"` | Plain links, SSR-first pages, no framework wrapper    |
| `render={<Link />}`    | Next.js, TanStack Router, Remix, or custom router API |
| `render={<button />}`  | Local state, client-side tables, or filtered lists    |

Prefer `href` when navigation is the main job and the platform link is enough. Prefer `render` with
a router `Link` when the app already relies on framework-level prefetching, active-state handling,
or typed route params.

### Next.js

Use `render` with `next/link` when the app standardizes on framework links.

```tsx
import {
  Pagination,
  PaginationContent,
  PaginationEllipsis,
  PaginationItem,
  PaginationLink,
  PaginationNext,
  PaginationPrevious,
  usePagination,
} from 'moduix';
import Link from 'next/link';
import { usePathname, useSearchParams } from 'next/navigation';

export function PaginationNextJs({ pageCount }: { pageCount: number }) {
  const pathname = usePathname();
  const searchParams = useSearchParams();
  const page = Number(searchParams.get('page') ?? '1');
  const pagination = usePagination({ count: pageCount, page });

  const createHref = (nextPage: number) => {
    const params = new URLSearchParams(searchParams);
    params.set('page', String(nextPage));
    return `${pathname}?${params.toString()}`;
  };

  return (
    <Pagination>
      <PaginationContent>
        <PaginationItem>
          <PaginationPrevious
            render={<Link href={createHref(pagination.previousPage)} />}
            aria-disabled={!pagination.canPreviousPage || undefined}
          />
        </PaginationItem>
        {pagination.items.map((item, index) => (
          <PaginationItem key={String(item) + '-' + index}>
            {typeof item !== 'number' ? (
              <PaginationEllipsis />
            ) : (
              <PaginationLink
                render={<Link href={createHref(item)} />}
                isActive={item === pagination.page}
              >
                {item}
              </PaginationLink>
            )}
          </PaginationItem>
        ))}
        <PaginationItem>
          <PaginationNext
            render={<Link href={createHref(pagination.nextPage)} />}
            aria-disabled={!pagination.canNextPage || undefined}
          />
        </PaginationItem>
      </PaginationContent>
    </Pagination>
  );
}
```

### TanStack Start

With TanStack Start, `render` can target router `Link` components directly.

```tsx
import {
  Pagination,
  PaginationContent,
  PaginationEllipsis,
  PaginationItem,
  PaginationLink,
  PaginationNext,
  PaginationPrevious,
  usePagination,
} from 'moduix';
import { Link, useSearch } from '@tanstack/react-router';

export function PaginationTanStackStart({ pageCount }: { pageCount: number }) {
  const search = useSearch({ from: '/' });
  const page = search.page ?? 1;
  const pagination = usePagination({ count: pageCount, page });

  return (
    <Pagination>
      <PaginationContent>
        <PaginationItem>
          <PaginationPrevious
            render={
              <Link
                to="/"
                search={(previous) => ({
                  ...previous,
                  page: pagination.previousPage,
                })}
              />
            }
            aria-disabled={!pagination.canPreviousPage || undefined}
          />
        </PaginationItem>
        {pagination.items.map((item, index) => (
          <PaginationItem key={String(item) + '-' + index}>
            {typeof item !== 'number' ? (
              <PaginationEllipsis />
            ) : (
              <PaginationLink
                render={
                  <Link
                    to="/"
                    search={(previous) => ({
                      ...previous,
                      page: item,
                    })}
                  />
                }
                isActive={item === pagination.page}
              >
                {item}
              </PaginationLink>
            )}
          </PaginationItem>
        ))}
        <PaginationItem>
          <PaginationNext
            render={
              <Link
                to="/"
                search={(previous) => ({
                  ...previous,
                  page: pagination.nextPage,
                })}
              />
            }
            aria-disabled={!pagination.canNextPage || undefined}
          />
        </PaginationItem>
      </PaginationContent>
    </Pagination>
  );
}
```
